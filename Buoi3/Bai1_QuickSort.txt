// quick sort
#include <stdio.h>
#include <iostream>
#define SIZE 100

void input(int A[], int &n)
{
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", &A[i]);
}

void output(int A[], int n)
{
	for (int i = 0; i < n; i++)
		printf("%d ", A[i]);
}

void swap(int &x, int &y)
{
	int temp = x;
	x = y;
	y = temp;
}

void quickSort(int A[], int left, int right)
{
	int pivot = A[(left + right) / 2]; // bien cam canh
	int i = left, j = right;
	// buoc phan hoach
	while (i <= j)
	{
		while (A[i] < pivot) // tim vi tri sai
			i++;
		while (A[j] > pivot) // tim vi tri sai
			j--;
		if (i <= j)
		{
			swap(A[i], A[j]);
			i++;
			j--;
		}
	}
	// buoc de qui
	if (left < j)
		quickSort(A, left, j);
	if (right > i)
		quickSort(A, i, right);
}

int main()
{
	int A[SIZE], n;
	input(A, n);
	int left = 0, right = n - 1;
	quickSort(A, left, right);
	output(A, n);
	system("pause");
	return 0;
}
